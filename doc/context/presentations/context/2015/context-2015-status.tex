\starttexdefinition statusstep #1
    \startparagraph
        #1
    \stopparagraph
\stoptexdefinition

\starttexdefinition status #1#2
    \starttitle[title={#1}]
        \processcommalist[#2]\statusstep
    \stoptitle
\stoptexdefinition
