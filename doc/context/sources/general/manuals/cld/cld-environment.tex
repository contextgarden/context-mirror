% special functions

\unexpanded\def\ShowLuaExampleOne#1#2#3%
  {\bgroup
   \obeyluatokens
   \startsubsummary[title={#1.#2(#3)}]
   \ctxlua{table.tocontext(#3)}
   \stopsubsummary
   \egroup}
