%D Some shortcuts.

% ((#1(type\ |)chapter#2)|#3)

\def\insertchap#1#2#3%
  {\insertchapter
    {#1\insertbibtype{}{\ }{capitolo\ }}{#2}%
    {#3}}

% #1city, country: pubname#2
% #1country: pubname#2
% #1pubname#2
% #1city, country#2
% #3

\def\insertpublisher#1#2#3%
   {\insertpubname
       {\insertcity
            {#1}
            {\insertcountry{, }{}{}: }%
            {#1\insertcountry{}{: }{}}}%
       {#2}%
       {\insertcity
            {#1}
            {\insertcountry{, }{}{#2}}%
            {\insertcountry{#1}{#2}{#3}}}%
    }

\def\insertorg#1#2#3%
   {\insertorganization
       {\insertcity
            {#1}
            {\insertcountry{, }{}{#2}: }%
            {\insertcountry{}{: }{#2}}}%
       {}%
       {\insertcity
            {#1}
            {\insertcountry{, }{}{#2}}%
            {\insertcountry{}{#2}{#3}}}%
    }
